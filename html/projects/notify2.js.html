INFO Class With Confirmation Example


var notify = new Notify2("Are you sure you want to do this?!", "info");
notify.doConfirm("confirmYes()", "confirmNo()");
notify.notify();

function confirmYes()
{
    notify = new Notify2("You chose yes!", "success");
    notify.notify();
}
function confirmNo()
{
    notify = new Notify2("You chose no! OH NO!!!", "warn");
    notify.notify();
}
